# 이벤트 핸들러
이벤트 발생 시 해당 이벤트를 처리하는 함수 (= 이벤트 리스너)

## 이벤트 핸들러 추가
```js
    //클래스 컴포넌트 안
    constructor(props){
        this.handleClick = this.handleClick.bind(this) //callback에서 this 사용을 위해서는 바인딩을 꼭 해주어야 한다. (기본적으로 바인드되지 않음)

    }

    handleClick() {
        this.setState(prevState => ({
            isToggleOn: !prevState.isToggleOn
        }));
    }

    //함수 컴포넌트에서는...
    //1. 함수 안 함수로 정의
    function handleClick() {
        setIsToggleOn((isToggleOn) => !isToggleOn);
    }

    //2. arrow function 이용
    const handleClick = () => {
        setIsToggleOn((isToggleOn) => !isToggleOn);
    }
```

## 매개변수 전달
```js
    <button onClick={(event) => handleDelete(1, event)}>삭제하기</button>
```
